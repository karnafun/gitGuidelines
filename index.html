פעולות בסיסיות עם GIT

יש להריץ את הפקודות דרך git bash

git pull origin master

הוספת הקבצים לסביבת staging
git add *
ביצוע קומיט לקבצים שהוספנו
git commit -m "My Commit Message"
דחיפת הקומיטים המקומיים שבוצעו לתוך השרת
git push
אם מדובר בענף חדש וזו הפעם הראשונה שדוחפים אליו מידע יש לכתוב
git push -u origin BRANCHNAME

git checkout BRANCHNAME
אם רוצים ליצור ענף חדש ולעבור אליו יש להוסיף את הדגל -b
git checkout -b BRANCHNAME

אם git status מראה שינויים באדום שאתם רוצים למחוק
git checkout .
הפעולה הזאת בעצם עוברת לענף שבו אתם נמצאים, עם הגרסא הלוקלית שלו (אשר לא כוללת את השינויים שלא בstage)
git checkout -b BRANCHNAME

אם התווספו קבצים חדשים שלא נמצאים עדיין בgit ניתן להסיר אותם
git clean -fd

שלב ראשון אם עדיין לא בוצע, יש לבצע קומיט לכל השינויים בשביל שלא נאבד מידע
git add *
git commit -m 'COMMIT MESSAGE'
עכשיו ניתן ליצור ענף חדש ולעבור אליו
git checkout -b NEWBRANCHNAME
ולאחר המעבר לדחוף את השינויים
git push -u NEWBRANCHNAME
אם נרצה לאחר מכן להחזיר את הענף הקודם שיהיה זהה לגרסא בשרת
git fetch origin
git reset --hard origin/BRANCHNAME
